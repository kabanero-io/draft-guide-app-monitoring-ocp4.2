cat << EOF | oc create -n "myapp" -f -
apiVersion: extensions/v1beta1
kind: Deployment
metadata:
  name: example-app
spec:
  replicas: 3
  template:
    metadata:
      labels:
        app: example-app
    spec:
      containers:
      - name: example-app
        image: fabxc/instrumented_app
        ports:
        - name: web
          port: 9090
          protocol: TCP
          targetPort: web
EOF



cat << EOF | oc create -n "myapp" -f -
kind: Service
apiVersion: v1
metadata:
  name: example-app
  labels:
    app: example-app
    k8s-app: myapp
spec:
  selector:
    app: example-app
  ports:
  - name: web
    port: 9090
    protocol: TCP
    targetPort: webs
EOF
